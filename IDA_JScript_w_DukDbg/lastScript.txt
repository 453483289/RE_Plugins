
//00AC717C   55               PUSH EBP
//00ACAD55   C3               RETN

ea = 0x0A4717C
while(ea < 0x0A4AD55 ){

    asm = ida.isCode(ea);
    if(asm == 1){
        sz = ida.instSize(ea);
    }else{
        ida.makeCode(ea);
        asm = ida.isCode(ea);
        if(asm == 1){
            sz = ida.instSize(ea);  
        }else{
            sz = 1;
        }
    }
    ea += sz;

}
